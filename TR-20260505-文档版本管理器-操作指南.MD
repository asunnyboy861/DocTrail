### 2.2 重点项目深度分析

#### 项目A：automerge/automerge-swift（核心同步引擎）

**为什么选它**: Automerge是CRDT（Conflict-free Replicated Data Type）库，天然解决多人协作冲突问题——这正是文档版本管理的核心技术难点。

**核心能力**:
- 自动合并冲突：多个用户同时编辑，自动合并无需手动解决
- 离线优先：本地操作，联网后自动同步
- 跨平台：iOS/macOS/Web可互操作
- 版本历史：完整的变更追踪和时间旅行

**二次开发方式**: 作为Swift Package集成，用于文档协作的冲突解决层。

#### 项目B：mentalfaculty/LLVS（版本化存储引擎）

**为什么选它**: LLVS专门为版本化数据设计，提供Git-like的分支、合并、差异比较功能，且原生支持CloudKit同步。

**核心能力**:
- 每次保存自动创建版本
- 版本分支、合并、差异比较
- CloudKit同步交换
- 多设备同步无需网络代码
- 三路合并算法

**二次开发方式**: 作为文档版本的底层存储引擎，替代自建版本管理系统。

#### 项目C：glushchenko/fsnotes（UI/文件管理参考）

**为什么选它**: 7310星，成熟的macOS/iOS文件管理应用，包含Git版本集成、iCloud同步、加密等完整功能。

**可借鉴部分**:
- 文件列表UI布局

#### 3.3.2 版本管理引擎

```swift
import Foundation
import CryptoKit

actor VersionEngine {
    private let fileManager = FileManager.default
    private let versionsDirectory: URL
    
    init(containerURL: URL) {
        self.versionsDirectory = containerURL.appendingPathComponent("DocTrailVersions", isDirectory: true)
        try? fileManager.createDirectory(at: versionsDirectory, withIntermediateDirectories: true)
    }
    
    func createVersion(
        for documentURL: URL,
        documentId: UUID,
        author: String,
        description: String,
        parentVersionId: UUID?
    ) async throws -> DocVersion {
        let data = try Data(contentsOf: documentURL)
        let hash = computeHash(data)
        let versionNumber = try await getNextVersionNumber(for: documentId)
        
        let version = DocVersion(
            documentId: documentId,
            versionNumber: versionNumber,
            createdBy: author,
            changeDescription: description,
            fileSize: Int64(data.count),
            fileHash: hash
        )
        version.parentVersionId = parentVersionId
        
        let versionFileURL = versionsDirectory
            .appendingPathComponent(documentId.uuidString, isDirectory: true)
            .appendingPathComponent("v\(versionNumber)")
        
        try fileManager.createDirectory(
            at: versionFileURL.deletingLastPathComponent(),
            withIntermediateDirectories: true
        )
        try data.write(to: versionFileURL)
        
        return version
    }
    
    func restoreVersion(_ version: DocVersion, to targetURL: URL) async throws {
        let versionFileURL = versionsDirectory
            .appendingPathComponent(version.documentId.uuidString, isDirectory: true)
            .appendingPathComponent("v\(version.versionNumber)")
        
        let data = try Data(contentsOf: versionFileURL)
        try data.write(to: targetURL)
    }
    
    func computeDiff(between oldVersion: DocVersion, and newVersion: DocVersion) async throws -> [ChangeItem] {
        let oldURL = versionsDirectory
            .appendingPathComponent(oldVersion.documentId.uuidString, isDirectory: true)
            .appendingPathComponent("v\(oldVersion.versionNumber)")
        let newURL = versionsDirectory
            .appendingPathComponent(newVersion.documentId.uuidString, isDirectory: true)
            .appendingPathComponent("v\(newVersion.versionNumber)")
        
        let oldData = try Data(contentsOf: oldURL)
        let newData = try Data(contentsOf: newURL)
        
        var changes: [ChangeItem] = []
        
        if oldData.count != newData.count {
            changes.append(ChangeItem(
                type: .modified,
                field: "fileSize",
                oldValue: "\(oldData.count)",
                newValue: "\(newData.count)"
            ))
        }
        
        let oldHash = computeHash(oldData)
        let newHash = computeHash(newData)
        if oldHash != newHash {
            changes.append(ChangeItem(
                type: .modified,
                field: "content",
                oldValue: oldHash,
                newValue: newHash
            ))
        }
        
        return changes
    }
    
    private func computeHash(_ data: Data) -> String {
        let digest = SHA256.hash(data: data)
        return digest.compactMap { String(format: "%02x", $0) }.joined()
    }
    
    private func getNextVersionNumber(for documentId: UUID) async throws -> Int {
        let docDir = versionsDirectory.appendingPathComponent(documentId.uuidString, isDirectory: true)
        guard let contents = try? fileManager.contentsOfDirectory(at: docDir, includingPropertiesForKeys: nil) else {
            return 1
        }
        let versionNumbers = contents.compactMap { url -> Int? in
            guard let name = url.pathComponents.last, name.hasPrefix("v") else { return nil }
            return Int(name.dropFirst())
        }
        return (versionNumbers.max() ?? 0) + 1
    }
}
```

#### 3.3.3 CloudKit同步引擎

```swift
import CloudKit

actor CloudKitSyncEngine {
    private let container: CKContainer
    private let privateDB: CKDatabase
    private let sharedDB: CKDatabase
    private let zoneID: CKRecordZone.ID
    
    init(containerIdentifier: String) {
        self.container = CKContainer(identifier: containerIdentifier)
        self.privateDB = container.privateCloudDatabase
        self.sharedDB = container.sharedCloudDatabase
        self.zoneID = CKRecordZone.ID(zoneName: "DocTrail", ownerName: CKCurrentUserDefaultName)
    }
    
    func setupZone() async throws {
        let zone = CKRecordZone(zoneID: zoneID)
        let operation = CKModifyRecordZonesOperation()
        operation.recordZonesToSave = [zone]
        try await privateDB.modifyRecordZones(saving: [zone], deleting: [])
    }
    
    func pushVersion(_ version: DocVersion, fileData: Data) async throws {
        let record = CKRecord(recordType: "DocVersion", recordID: CKRecord.ID(recordName: version.id.uuidString, zoneID: zoneID))
        
        record["documentId"] = version.documentId.uuidString
        record["versionNumber"] = version.versionNumber
        record["createdBy"] = version.createdBy
        record["changeDescription"] = version.changeDescription
        record["fileSize"] = version.fileSize
        record["fileHash"] = version.fileHash
        record["createdAt"] = version.createdAt
        
        let asset = CKAsset(fileData: fileData)
        record["fileData"] = asset
        
        try await privateDB.save(record)
    }
    
    func pullVersions(for documentId: UUID) async throws -> [DocVersion] {
        let predicate = NSPredicate(format: "documentId == %@", documentId.uuidString)
        let query = CKQuery(recordType: "DocVersion", predicate: predicate)
        query.sortDescriptors = [NSSortDescriptor(key: "versionNumber", ascending: true)]
        
        let (matchResults, _) = try await privateDB.records(matching: query, inZoneWith: zoneID)
        
        var versions: [DocVersion] = []
        for (_, result) in matchResults {
            guard let record = try? result.get() else { continue }
            let version = DocVersion(
                documentId: UUID(uuidString: record["documentId"] as? String ?? "") ?? UUID(),
                versionNumber: record["versionNumber"] as? Int ?? 0,
                createdBy: record["createdBy"] as? String ?? "Unknown",
                changeDescription: record["changeDescription"] as? String ?? "",
                fileSize: record["fileSize"] as? Int64 ?? 0,
                fileHash: record["fileHash"] as? String ?? ""
            )
            versions.append(version)
        }
        return versions
    }
    
    func shareDocument(documentId: UUID, with userEmails: [String], permission: CKShare.Participant.Role) async throws {
        let share = CKShare(rootRecord: CKRecord(recordType: "Document", recordID: CKRecord.ID(recordName: documentId.uuidString, zoneID: zoneID)))
        share[CKShare.SystemFieldKey.title] = "DocTrail Shared Document"
        share[CKShare.SystemFieldKey.shareType] = "com.doctrail.share"
        
        try await privateDB.save(share)
    }
}
```

#### 3.3.4 文件监控器

```swift
import Foundation

class FileWatcher: ObservableObject {
    private var source: DispatchSourceFileSystemObject?
    private let queue = DispatchQueue(label: "com.doctrail.filewatcher", qos: .utility)
    
    @Published var detectedChanges: [FileChange] = []
    
    struct FileChange {
        let url: URL
        let type: ChangeType
        let timestamp: Date
        
        enum ChangeType {
            case created, modified, deleted, renamed, moved
        }
    }
    
    func startWatching(directory: URL) {
        let descriptor = open(directory.path, O_EVTONLY)
        guard descriptor >= 0 else { return }
        
        source = DispatchSource.makeFileSystemObjectSource(
            fileDescriptor: descriptor,
            eventMask: [.write, .delete, .rename],
            queue: queue
        )
        
        source?.setEventHandler { [weak self] in
            self?.handleFileChange(at: directory)
        }
        
        source?.setCancelHandler {
            close(descriptor)
        }
        
        source?.resume()
    }
    
    func stopWatching() {
        source?.cancel()
        source = nil
    }
    
    private func handleFileChange(at directory: URL) {
        DispatchQueue.main.async {
            self.detectedChanges.append(FileChange(
                url: directory,
                type: .modified,
                timestamp: Date()
            ))
        }
    }
}
```

### 3.4 编写规则

| 规则 | 说明 |
|------|------|
| 最低支持版本 | iOS 17.0 / iPadOS 17.0 / macOS 14.0 |
| 并发模型 | 全部使用 Swift Concurrency (async/await, Actor) |
| 数据持久化 | SwiftData 为主，Core Data 仅用于 CloudKit 兼容 |
| UI范式 | SwiftUI + @Observable，禁止使用 UIKit 除非必要 |
| 错误处理 | 使用 typed throws，自定义 DocTrailError 枚举 |
| 命名规范 | 类型用 PascalCase，方法用 camelCase，常量用 PascalCase |
| 文件组织 | 每个文件一个主要类型，扩展放在同目录 Extensions 子文件夹 |
| 网络层 | 原生 URLSession + AsyncAwait，不使用第三方网络库 |
| 安全 | 所有文件操作使用 App Sandbox，CloudKit 使用用户私有数据库 |
| 测试 | 每个核心 Engine 必须有 Unit Test，UI 用 XCUITest |

## 七、极具竞争力的价格策略

### 7.1 定价策略分析

| 策略类型 | 适用场景 | 本项目适配度 | 分析 |
|----------|----------|-------------|------|
| 完全免费 | 超简单引流工具 | ❌ 不适合 | 版本管理需要存储和同步成本，完全免费不可持续 |
| 下载前付费 | 简单软件 | ❌ 不适合 | B2B用户需要先体验才能判断价值，付费门槛过高 |
| 站内订阅A | 功能简单+有API费用 | ⚠️ 部分适合 | CloudKit无额外费用，但团队功能需要持续维护 |
| **站内订阅B** | 功能稍复杂+免费下载+免费功能+按月/按年/一次性买断 | ✅ **最适合** | 符合B2B SaaS定价模式，免费层引流，订阅变现 |

### 7.2 最终定价方案：站内订阅B

**选择理由**:
1. B2B用户习惯"先试后买"，免费下载降低获客门槛
2. 团队协作功能天然适合订阅模式（持续价值=持续付费）
3. CloudKit虽无API费用，但团队功能（共享数据库、推送通知）有持续维护成本
4. 小团队对$5-15/月价格敏感度低，对$50+/月敏感度高

## 十一、技术风险与缓解
